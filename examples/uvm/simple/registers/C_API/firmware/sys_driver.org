#include "sys_driver.h"

using namespace snps_reg;

int
sys_driver::trivial(sys_t Sys)
{
    unsigned int rd;
    rd = regRead(Sys.Comp1a.regA());
    rd = regRead(Sys.Comp1a.regB_fldA());
    rd = regRead(Sys.Comp1b.regB_fldA());
    rd = regRead(Sys.Comp1b.regC());
    rd = regRead(Sys.Comp2.regA());
    rd = regRead(Sys.Comp2.rf1[0].r0());
    rd = regRead(Sys.Comp2.rf1[3].r1_fB());
    rd = regRead(Sys.Comp2.rf1[3].r2());
    rd = regRead(Sys.Comp2.regE());
    
    regWrite(Sys.Comp2.regE(), 0x1111);
    regWrite(Sys.Comp1a.regB(), 0x2222);
    regWrite(Sys.Comp1b.regC(), 0x3333);
    regWrite(Sys.Comp1b.regB_fldC(), 0x4444);
    regWrite(Sys.Comp2.rf1[3].r1_fA(), 0x5555);
    regWrite(Sys.Comp2.regE(), 0x1111);
    
    return 0;
};


int
sys_driver::comp1_drv(comp1_t dev)
{
    unsigned int rd;
    rd =  regRead(dev.regA());
    regWrite(dev.regB(), 0xFAFAFA);
#ifdef SNPS_UVMC_ONLY
    rd =  regGet(dev.regA());
    regSet(dev.regB(), 0xFAFAFA);
#endif
    return 0;
};
